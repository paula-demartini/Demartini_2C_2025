117void distribuye_bits (int8_t numero, struct GPIO_DATA *vector) { //numero ya está en BCD
118
119/*vector es un puntero al vector con 4 estructuras del tipo GPIO que guardan el pin y el estado de dicho pin (in/out)*/

121 int8_t filtro = 00000001; //para la máscara (AND)
122
123 for (int i = 0; i<4 ; i++) {

125 int b = numero & filtro; //en b queda el bit de interés (analiza de a uno)
126 numero = numero/2; //corre los bits de numero a la derecha, rellenando con ceros el MSB

146void coordina_displays (int32_t data, int8_t digits, struct GPIO_DATA *vector1, struct GPIO_DATA *vector2) {
147
148/*vector1 es vector y vector2 es vector_LCD*/
149
150 uint8_t ArregloBCD[digits]; //define un vector vacío de enteros con "digits" elementos
151 conversor_a_BCD (data, digits, ArregloBCD); //convierte a BCD y guarda cada dígito en ArregloBCD
152
153 for (int i = 0; i<digits; i++) {
154
155 distribuye_bits (ArregloBCD[i], vector1); //distribuye los 4 bits de un dígito en las 4 entradas del display
156 GPIOOn(vector2[i].pin); //lo prende y lo apaga; le da un pulso al display correspondiente para encenderlo
